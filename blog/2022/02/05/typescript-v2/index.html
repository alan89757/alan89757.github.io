# 什么是typescript

// 定义泛型
function getValue<T>(arg: T): T {
  return arg;
}

// 2. 利用typescript类型推断, 推断为string类型

// 泛型约束 extends
function getLength<T>(arg: T): T {
  console.log(arg.length);   // 类型“T”上不存在属性“length”, 没有泛型约束不能这样写
  return arg;
}

// 定义接口
interface Lengthwise {
  length: number
}
// extends关键字对泛型约束
function getLength2<T extends Lengthwise>(arg: T): T { // extends对泛型约束
  console.log(arg.length); // 泛型约束后不会报错, 只允许传入包含length属性的变量
  return arg;
}
// 使用泛型
const num = getLength2(1); // 没有length属性. 报错:类型“number”的参数不能赋给类型“Lengthwise”的参数
const str = getLength2('hello');  // 字符串有length属性
const arr = getLength2([1,2,3]);  // 数组有length属性
const obj = getLength2({length: 3}); // 对象有length属性

  age: number;
}
const p1: Partial<Person1> = {};  // Person1中所有属性变成可选属性了

// 3. Exclude<T, U>
type T0 = Exclude<'a' | 'b' | 'c', 'a'>  // 'b' | 'c'
type T1 = Exclude<"a" | "b" | "c", "a" | "b">; // "c"
type T2 = Exclude<string | number | (()=>void), Function>; // string | number

// 4. Extract 和 Exclude 相反，Extract<T,U> 从 T 中提取出 U
type T3 = Extract<"a" | "b" | "c", "a" | "f">; // "a"
type T4 = Extract<string | number | (() => void), Function>; // () =>void

// 5. Readonly 把数组或对象的所有属性值转换为只读的
interface Person {
  name: string;
  age: number;
  gender?: "male" | "female";
}

let p: Readonly<Person> = {
  name: "hello",
  age: 10,
  gender: "male",
};
p.age = 20; // 无法为“age”赋值，因为它是只读属性

// 6. Record Record<K extends keyof any, T> 的作用是将 K 中所有的属性的值转化为 T 类型
type Property = 'key1' | 'key2';
type Person = Record<Property, string>
const p: Person = {
  key1: 'alan',
  key2: 'hello'
}

// 7. 从某个类型中挑出一些属性出来
type Person1 = {
  name: string;
  age:number;
  gender:string
}
type P1 = Pick<Person1, 'name' | 'age'>;
const user: P1 = {
  name: 'alan',
  age: 20
}

// 8. Omit 与Pick相反，Omit<T,K> 从T中取出除去K的其他所有属性。
interface Person2 {
  name: string,
  age: number,
  gender: string,
}
type P2 = Omit<Person2, 'age'|'gender'>
const user2: P2 = {
  name: 'alan',
}

// 9. NonNullable 去除类型中的 null 和 undefined
type P1 = NonNullable<string|number|undefined>; // string | number
type P2 = NonNullable<string[] | null | undefined>;  // string[]

// 10. ReturnType 用来得到一个函数的返回值类型
type Func = (value: string)=> number;
const test: ReturnType<Func> = 1;

// 11. Parameters 用于获得函数的参数类型所组成的元组类型。
type P1 = Parameters<(a: number, b: string)=>void>; // [number, string]

// 12. InstanceType 返回构造函数类型T的实例类
class C {
  x = 0;
  y = 0;
}
type D = InstanceType<typeof C>;  // C
