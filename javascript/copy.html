# 实现深浅拷贝

# 数组的浅拷贝

可以直接利用数组的一些方法实现拷贝，代码示例如下：

var arr = ["old", 1, "a"];
var arr_new = arr.concat();
var arr_new2 = arr.slice();
arr_new[0] = "new";
console.log(arr);   // ["old", 1, "a"];
arr_new2[0] = "new2";
console.log(arr);  // ["old", 1, "a"];

如果数组内有引用数据类型，代码示例如下：

var arr1 = [{a: "old"}, 1, "a", ];
var arr_new = arr1.concat();
arr_new[0]["a"] = "new";
console.log(arr1);   // [{a: "new"}, 1, "a"];

var arr2 = [{b: "old"}, 1, "a", ];
var arr_new2 = arr2.slice();
arr_new2[0]["b"] = "new2";
console.log(arr2);   // [{b: "new2"}, 1, "a"];

结论：数组中有引用类型的数据，使用concat，slice拷贝会互相影响，所以它们是一种浅拷贝。

# 数组的深拷贝

JSON.stringify支持深拷贝，但函数会被过滤掉

var arr1 = [{a: "old"}, [1,2,3], 1 ];
var arr_new = JSON.stringify(arr1);
arr_new[0]["a"] = "new";
console.log(arr1);   // [{a: "old"}, 1, "a"];

如果数组对象中有函数

 var arr = [
    function fn(){
        console.log(1);
    },
    {
        b: function() {
            console.log(2);
        }
    }
];
var new_arr = JSON.parse(JSON.stringify(arr));
console.log(new_arr);  // [null, {}]

# 实现浅拷贝

抛开数组技巧类，自己实现浅拷贝

思路：遍历对象，把属性和属性值都放到一个新对象中，代码示例如下：

 // 实现浅拷贝
function shallowCopy(obj) {
    if(typeof obj !== "object") { // 基本数据类型直接返回该值
        return obj;
    }
    // 根据数据类型新建数组或者对象
    var newObj = obj instanceof Array ? [] : {};
    // 遍历obj，只有obj自己的属性才复制
    for (const key in obj) {
        if (obj.hasOwnProperty(key)) {
            newObj[key] = obj[key];
        }
    }
    return newObj;
}

var arr = ["old", 1, "a"];
arr_new = shallowCopy(arr);
arr_new[0] = "new";
console.log(arr);  // ['old', 1, 'a']
console.log(arr_new);  // ['new', 1, 'a']

# 深拷贝的简单实现

思路：如果是对象就递归调用浅拷贝函数~

 // 深拷贝简单实现
function deepCopy(obj) {
    if(typeof obj !== "object") { // 基本数据类型直接返回该值
        return obj;
    }
    // 根据数据类型新建数组或者对象
    var newObj = obj instanceof Array ? [] : {};
    // 遍历obj，只有obj自己的属性才复制
    for (const key in obj) {
        if (obj.hasOwnProperty(key)) {
            if(typeof obj[key] === "object") { // 是对象就递归调用浅拷贝函数
                newObj[key] = deepCopy(obj[key]);
            } else {
                newObj[key] = obj[key];
            }
        }
    }
    return newObj;
}
var arr = [{a: "old"}, [1,2,3], 1 ];
arr_new = deepCopy(arr);
arr_new[0]["a"] = "new";
console.log(arr);  // [{a: "old"}, [1,2,3], 1 ];
console.log(arr_new);  // [{a: "new"}, [1,2,3], 1 ];

 // 思路： 根据参数个数判断
// 1. 只有一个参数，是给jQuery函数对象或者函数原型添加属性或方法， 通过this就可以获取当前是函数对象或者实例对象
// 2. 把所有的对象赋值到第一个参数对象上
function extend() {
    var target;  // 目标对象
    var i = 1;  // arguments[0]要保存为target对象，遍历从i=1开始
    var name; 
    var length = arguments.length;  // 获取实参长度
    target = arguments[0];   // 获取第一个参数
    
    if(typeof target !== "object") {
        target = {};
    }
    // i从1开始
    for (; i < length; i++) {
        var args = arguments[i];
        for (name in args) {
            if (args.hasOwnProperty(name)) {
                target[name] = args[name];
            }
        }
    }
    return target;  
}


--------------------------------extend深拷贝完整版(jquery.js)-------------------------------------------
(function(root) {
    var jQuery = function() {
        return new jQuery.prototype.init();
    }
    jQuery.fn = jQuery.prototype = {
        init: function() {

        }
    }

    jQuery.prototype.init.prototype = jQuery.prototype;  // 把jQuery.prototype.init匿名函数的原型指向jQuery的原型
    // 思路： 根据参数个数判断
    // 1. 只有一个参数，是给jQuery函数对象或者函数原型添加属性或方法， 通过this就可以获取当前是函数对象或者实例对象
    // 2. 把所有的对象赋值到第一个参数对象上
    jQuery.extend = jQuery.prototype.extend = function() {
        var target = arguments[0] || {};  // 目标对象，赋值第一个参数
        var i = 1;  // arguments[0]要保存为target对象，遍历从i=1开始
        var name; // 遍历键值
        var deep = false;  // 是否深拷贝
        var option; // 保存待复制对象
        var copy;  // 待复制对象的值
        var copyIsArray;
        var clone;
        var length = arguments.length;  // 获取实参长度
        
        if(typeof target === "boolean") {  // 为true就是深拷贝
            deep = target;
            target = arguments[1];
            i++;                // i++把for循环遍历后移一位
        }
        if(typeof target !== "object") { // 设置默认值
            target = {};
        }
        if(length === i) { // 只有一个参数
            target = this;  // 把当前上下文（实例对象或者函数对象）复制给target目标对象
            i--;
        }
        // i从1开始，遍历从1开始~
        for (; i < length; i++) {
            if((option = arguments[i]) !== null) {
                for (name in option) {
                    src = target[name];  // 目标对象
                    copy = option[name];  // 待复制对象
                    if(deep && ((copyIsArray=$.isArray(copy)) || $.isPlainObject(copy))) { // 对象或者数组引用数据类型需要递归，赋值要(copyIsArray=$.isArray(copy))用括号包裹起来
                        if(copyIsArray) { // 是数组，待复制对象决定目标对象的数据类型和默认值
                            copyIsArray = false;  // 重置下copyIsArray为false
                            clone = src && $.isArray(src) ? src : [];       // 目标对象跟着待复制对象的类型走
                        } else {
                            clone = src && $.isPlainObject(src) ? src : {}; 
                        }
                        target[name] =  $.extend(deep, clone, copy);  // 这要赋值给target目标对象
                    } else {
                        if (copy !== undefined) {
                            target[name] = copy;
                        }
                    }
                }
            }
        }
        return target;  
    }

    jQuery.extend({
       isArray: function(obj) { // 判断是否数组
            return Object.prototype.toString.call(obj) === "[object Array]";
       },
       isPlainObject: function(obj) { // 纯对象
            return typeof obj === "object";
       }
    })
    root.jQuery = root.$ = jQuery;
})(window)
