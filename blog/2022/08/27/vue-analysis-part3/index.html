// 没有编译器, attr in obj或者with作用域, 获取对象属性会触发这个钩子, 所以可以做「过滤器」
// 会被编译成 with(this){return _c('div',{attrs:{"id":"app"}},[_v(_s(message)+_s(_test))])}
var hasHandler = {
  has: function has() { 
    var has = key in target;
    var isAllowed = allowedGlobals(key) ||
        (typeof key === 'string' && key.charAt(0) === '_' && !(key in target.$data));
    return has || !isAllowed;
  }
}
// 代理函数
var hasProxy =
      typeof Proxy !== 'undefined' && isNative(Proxy);
var initProxy = function initProxy(vm) {
  if (hasProxy) {
    var options = vm.$options;
    var handlers =
      options.render && options.render._withStripped ? getHandler : hasHandler;
    // 代理vm实例到vm属性_renderProxy，做数据过滤
    vm._renderProxy = new Proxy(vm, handlers);
  } else {
    vm._renderProxy = vm;
  }
};

// render函数执行完结果
// vm.$options._render
ƒ anonymous() {
    with(this){return _c('div',{attrs:{"id":"app"}},[_v(_s(message)+_s(_test))])}
}

# 数据过滤

var hasHandler = {
  has: function has (target, key) {
    var has = key in target;
    // isAllowed用来判断模板上出现的变量是否合法。
    var isAllowed = allowedGlobals(key) ||
        (typeof key === 'string' && key.charAt(0) === '_' && !(key in target.$data));
    // _和$开头的变量不允许出现在定义的数据中，因为他是vue内部保留属性的开头。
    if (!has && !isAllowed) {
      // 1. 警告不能以$ _开头的变量
      // 2. 警告模板出现的变量在vue实例中未定义
    }
    return has || !isAllowed
  }
};
