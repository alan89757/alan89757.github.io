分析nextTick执行机制

# 应用场景

export default {
  data () {
    return {
      count: 0
    }
  },
  mounted () {
    this.count = 1
    this.count = 2
    this.count = 3
  },
  watch: {
    count () {
      console.log(this.count);   // 只输出一次 ‘3’
    }
  }
}

count更新了三次，理论上页面会渲染三次。但实际上只渲染一次。

总结：同一个watcher实例不会被重复添加到更新队列queue。代码实现如下：

var has = {};
function queueWatcher(watcher) {
  var watcherId = watcher.id; // 一个Watcher依赖只渲染一次
  if(!has[watcherId]) { // 同一个watcher实例不会被重复添加到更新队列
    has[watcherId] = true;
    queue.push(watcher);
    // ...
  } 
}

# flushSchedulerQueue

把flushSchedulerQueue函数（DOM渲染）推入callbacks队列

var callbacks = [];
callbacks.push(flushSchedulerQueue)

# vm.$nextTick

vm.$nextTick把回调函数tickCb推入callbacks队列

function tickCb() { 
  // 操作dom 
}
vm.$nextTick(tickCb);  // callbacks.push(tickCb)

# microTimerFunc

var p = Promise.resolve();
microTimerFunc = function() {
  p.then(flushCallbacks);
}

# 执行回调函数列表

var callbacks = [];
callbacks.push(flushSchedulerQueue);  // flushSchedulerQueue渲染页面

function tickCb(){ 
  // 操作dom
}
// vm.$nextTick(tickCb); => callbacks.push(tickCb)
callbacks.push(tickCb); 

function flushCallbacks() {
  pending = false;
  var copies = callbacks.slice(0);  // [flushSchedulerQueue, tickCb]
  // 确保页面渲染后， 再执行vm.$nextTick自定义回调
  callbacks.length = 0;
  for (var i = 0; i < copies.length; i++) {
    copies[i]();
  }
}
