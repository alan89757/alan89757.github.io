# 如何实现数组去重

# 双层循环

思路：定义一个新数组result存储结果，循环遍历result数组中没有说明该值唯一，把它添加到result数组中返回。实例代码如下：

function unique(arr) {
    var length = arr.length;
    var result = [];  // 存储结果新数组
    var resLength = 0;
    var j;
    for (let i = 0; i < length; i++) {
        resLength = result.length;
        for (j = 0; j < resLength; j++) {
            if(arr[i] === result[j]) {  // 如果有相同的值，直接跳出当前循环, 否则把数据push到结果数组result中
                break ;
            }
        }
        if(j == resLength) { // 最重要的步骤：j=== regLength说明arr[i]是唯一的
            result.push(arr[i]);  // 把数据push到结果数组result中
        }
    }
    return result;
}

# 使用indexOf简化循环遍历

function unique2(arr) {
    var length = arr.length;
    var result = [];  // 存储结果新数组
    for (let i = 0; i < length; i++) {
        if(result.indexOf(arr[i]) == -1) { // 不在result结果数组中说明该值唯一
            result.push(arr[i]);
        }
    }
    return result;
}

# 排序后去重

思路： 先排序，只需要比较相邻的数据是否相等

对已排序的数组有较大的性能优势

function unique3(arr) {
    var sortedArr = arr.concat().sort();  // 克隆一个新数组排序
    var seen;  // 保存i-1
    for (let i = 0; i < sortedArr.length; i++) {
        if(!i || seen != sortedArr[i] ) {  // sortedArr[0]先通过。sortedArr[i]保存为seen跟sortedArr[i+1]比较
        
        } else {
            sortedArr.splice(i, 1);     // 删掉重复的值
        }
        seen = sortedArr[i];
    }
    return sortedArr;
}

# 迭代器优化

思路：在去重的循环中加入处理函数（比如：忽略大小写）

// 加入迭代器
function unique5(arr, isSort, iteratee) {
    var result = [];
    var seen = [];
    for (let i = 0; i < arr.length; i++) {
        var computed = iteratee ? iteratee(arr[i], i, arr) : value;  // 在循环过程中加入迭代处理函数
        // iteratee传参都是按照（value，key，原数组），跟$.each，map等类似
        if(isSort) { // 数组已排序
            if(!i || seen != sortedArr[i] ) {  // sortedArr[0]先通过。sortedArr[i]保存为seen跟sortedArr[i+1]比较
                result.push(arr[i]);
            } else {
                sortedArr.splice(i, 1);     // 删掉重复的值
            }
            seen = sortedArr[i];
        } else if(iteratee) {  // 放在isSort判断后是因为已排序的数组去重性能很高
            if(seen.indexOf(computed) == -1) {
                seen.push(computed);  // seen中存储为了上面的检索判断
                result.push(arr[i]);     // result才是真正的搜集数据
            }
        } else {
            if(result.indexOf(arr[i]) == -1) { // 不在result结果数组中说明该值唯一
                result.push(arr[i]);
            }
        }
    }
    return result;
}

# 语法简写优化

es5中filter方法可以简化for循环，很酷, 实例代码如下：

function unique6(arr) {
    return arr.filter(function(item, index , array){
        return array.indexOf(item) === index; // 很巧妙，理论上indexOf找到的索引值应该是等于index的，不等于说明当前值是重复的
    });
}

# 其他实现

思路：用对象保存键值对去判断是否重复 typeof 区分出1和"1"

// 用对象的键值对去保存
var arr = [3, 1, 1, 2, '1', 5,  '1'];
console.log(unique9(arr)); // [1, "1",  2, 3, 5]
function unique9() {
    var obj = {};  // 保存键值对做判断
    return arr.concat().sort().filter(function(item, index, array) {
        return obj.hasOwnProperty(typeof item + item) ? false : (obj[typeof item + item] = true);
    });
}

function unique10(arr) {
    return Array.from(new Set(arr));
}
function unique11(arr) {
    return [...new Set(arr)];
}
// es6 map
function unique12(arr) {
    const seen = new Map();
    return arr.filter(function(item){
        return !seen.has(item) && seen.set(item, 1);
    })
}
