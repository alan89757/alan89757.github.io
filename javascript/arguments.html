# 类数组对象与arguments

# 类数组对象

  1. 从读写，获取长度，遍历角度分析~
var args;
function fn(name, age, sex) {
    args = arguments;
}
fn("name", "age", "sex");
// 读取
console.log(args[1]);  // "age"
// 获取长度
console.log(args.length);  // 3
// 遍历
for (let i = 0; i < args.length; i++) {
    console.log(args[i]);    
}

# 类数组调用数组方法

  1. 直接调用
var args;
    function fn(name, age, sex) {
        args = arguments;
    }
    fn("name", "age", "sex");
    args.push(1);    // args.push is not a function， 无法直接调用
  1. Function.call间接调用数组方法
     let { slice, splice, concat } = Array.prototype;  // 原型上的方法
        let { from } = Array;       // 构造函数的方法
        var arrayLike = {0: 'name', 1: 'age', 2: 'sex', length: 3 }
        // var result = slice.call(arrayLike);
        // var result = splice.call(arrayLike, 0);
        // var result = from(arrayLike);
        var result = concat.apply([], arrayLike);
        console.log(result);

# Arguments对象

Arguments只定义在函数体中。在函数体中，arguments指代该函数的Arguments对象

function foo(name, age) {
    console.log(arguments);
}
foo('name', 'age', 'sex');

包含函数的实参，函数实参的个数，callee 如上图所示

name, age, sex表示实参， length表示实参的个数 callee代表函数本身，使用场景：函数自调用，不用写死函数名称, 如下代码：

    var data = [];
    for (var i = 0; i < 3; i++) {
        (data[i] = function() {
            console.log(arguments.callee.i);  // 取的是匿名函数挂载的i
        }).i = i;   
    }
    data[0]();
    data[1]();
    data[2]();

# arguments和对应参数的绑定

function fn(name, age) {
    // 场景一，修改实参name的值, 观察arguemnts[0]值是否更新
    // name  = "new name";
    // console.log(arguments[0]);  // new name
    // 场景二， 修改arguments[0]的值，观察实参name值是否更新
    // arguments[0] = "new name2";
    // console.log(name);      // new name
    // 场景三，修改形参age的值，观察arguments[1]值是否更新
    // age = "new age";
    // console.log(arguments[1]);  // undefined
    // 场景四，修改arguments[1]的值，观察age值是否更新
    arguments[1] = "new age2";
    console.log(age);       // undefined
}
fn("name");
// 结论： 传入的参数，实参和arguments的值会共享。当没有传入时，实参与arguments值不会共享

结论： 传入的参数，实参和arguments的值会共享。当没有传入时，实参与arguments值不会共享

# 传递参数

将参数从一个函数传到另一个函数

function bar(name, age) {
    console.log(name, age);  // name1 age1
}
function foo() {
    bar.apply(this, arguments)
}
foo("name1", "age1");

# ES6函数内arguments转数组

function fn(...args) {
    console.log(args);  // name1 age1
}
fn("name1", "age1");

# 类数组

哪些操作会产生类数组：

  1. 函数体内Arguments对象
  2. document.getElementsByTagName等返回类数组

应用

  1. 参数不定长
  2. 函数柯里化
  3. 递归调用
  4. 函数重载 ...
