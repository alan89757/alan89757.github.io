# 前言

实现类型判断，参考jquery的实现

需求： 需求一. 判断字符串和数字等基本数据类型 (typeof) 需求二. 判断日期，数组，正则，错误类型 (Object.prototype.toString) 需求三. 判断纯粹对象（plainObject）, 空对象，window对象(自定义，如jquery的isPlainObject)

# 判断基本数据类型(typeof)

console.log(typeof undefined);  // undefined
console.log(typeof null); // null
console.log(typeof true); // boolean
console.log(typeof 123);       // number
console.log(typeof "123")  // string
console.log(typeof {})      // object

结论：typeof判断基本数据类型，除了Null和Object(null是原型链的终点，也是一个对象)，其他都能区分出来~

# 看个例子

// Array, Function, Date, RegExp, Error
console.log(typeof new Array());        // object
console.log(typeof new Function("a", "b", "return a + b;")); // function
console.log(typeof new Date());     // // object
console.log(typeof new RegExp(/a/));  // // object
console.log(typeof new Error("Uncatch TypeError"));  // // object

结论： typeof可以区分function

# Array, Function, Date, RegExp, Error等Object对象如何区分呢？（Object.prototype.toString）

Object.prototype.toString返回"[object class名称]" ，代码示例如下：

// Array, Function, Date, RegExp, Error
console.log(Object.prototype.toString.call(Array()));        // [object Array]
console.log(Object.prototype.toString.call(new Function("a", "b", "return a + b;"))); // [object Function]
console.log(Object.prototype.toString.call(new Date()));     // [object Date]
console.log(Object.prototype.toString.call(RegExp(/a/)));  // [object RegExp]
console.log(Object.prototype.toString.call(Error("Uncatch TypeError")));  // [object Error]
// 基本数据类型
console.log(Object.prototype.toString.call(undefined));  // [object Undefined]
console.log(Object.prototype.toString.call(null)); // [object Null]
console.log(Object.prototype.toString.call(true)); // [object Boolean]
console.log(Object.prototype.toString.call(123));       // [object Number]
console.log(Object.prototype.toString.call("123"));  // [object String]
console.log(Object.prototype.toString.call({}));      // [object Object]

// 不常用的复杂类型
console.log(Object.prototype.toString.call(Math));      // [object Math]
console.log(Object.prototype.toString.call(JSON));      // [object JSON]

// Arguments对象
function fn() {
    console.log(Object.prototype.toString.call(arguments));   // [object Arguments]
}
fn();

# 手写type函数判断数据类型

-----------------------------------手写type判断数据类型--------------------------------------
  // 保存toString的内部class和类型的映射对象
var classTypes = {};

// 这儿不把Null加进去，因为IE6，Object.prototype.toString都返回[object Object]
"Boolean Number String Function Array Date RegExp Object Error".split(" ").map(function(item, index) {
    classTypes["[object " + item + "]"] = item.toLowerCase();
});

// 实现类似typeof返回数据类型，支持判断Boolean Number String Function Array Date RegExp Object Error等类型
function type(obj) {
    // null和undefined在这儿区分
    if(obj == null) {
        return obj + "";
    }
    return typeof obj === "object" || typeof obj === "function" ? classTypes[Object.prototype.toString.call(obj)] || "object" : typeof obj;
}
// Array, Function, Date, RegExp, Error
console.log(type(new Array()));        // array
console.log(type(new Function("a", "b", "return a + b;"))); // function
console.log(type(new Date()));     // [object Date]
console.log(type(new RegExp(/a/)));  // regexp
console.log(type(new Error("Uncatch TypeError")));  // error
// 基本数据类型
console.log(type(undefined));  // undefined
console.log(type(null)); // null
console.log(type(true)); // boolean
console.log(type(123));       // number
console.log(type("123"));  // string
console.log(type({}));      // object

# 判断纯粹对象（plainObject）

定义：纯粹的对象就是通过{}或者new Object创建包括0或多个键值对的对象 使用场景：plainObject是为了和其他javascript对象（如：null，数组，宿主对象documents）等区分

先看以下jquery的实现效果：

console.log($.isPlainObject({}))；  // true
console.log($.isPlainObject(new Object()));   // true
console.log($.isPlainObject(Object.create(null)));   // true
console.log($.isPlainObject(Object.assign({a: 1}, {b: 2})));   // true
console.log($.isPlainObject(new Person("alan")));  // false
console.log($.isPlainObject(Object.create({})))     // false

function Person(name) {
    this.name  = name;
}

jquery.isPlainObject

    var classTypes = {}; // 存放toString映射结果
    var toString = classTypes.toString; // 相当于Object.prototype.toString
    var hasOwn = classTypes.hasOwnProperty;  // 相当于Object.prototype.hasOwnProperty
    function isPlainObject() {
        var proto, Ctor;
        // toString排除返回非[object Object]的数据
        if(!!obj || toString.call(obj) !== "[object Object]") { return false; }
        proto = Object.getPrototypeOf(obj);
        if(!proto) {   // 没有原型的对象是纯粹的对象
            return true; 
        }
        Ctor = hasOwn.call(proto, "constructor") && proto.constructor;  // 原型上有自己的构造函数就保存到Ctor
        // Ctor是函数 && 构造函数的原型是Object
        // 注意：toString是Funciton.protype原型上的toString，它重写了Object原型上的toString
        return typeof Ctor === "function" && hasOwn.toString.call(Ctor) === hasOwn.toString.call(Object);
    }

# 判断是否空对象（EmptyObject）

判断空对象，jquery源码如下：

function isEmptyObject(obj) {
    for (const key in obj) {
        if (obj.hasOwnProperty(key)) {
            return false;
        }
    }
    return true;
}
console.log(isEmptyObject(undefined));  // true
console.log(isEmptyObject(null));       // true
console.log(isEmptyObject(true));       // true
console.log(isEmptyObject(2));          // true
console.log(isEmptyObject(""));         // true
console.log(isEmptyObject({}));         // true
console.log(isEmptyObject([]));         // true

# isArrayLike

判断是否是数组或者类数组

var classTypes = {};
// 这儿不把Null加进去，因为IE6，Object.prototype.toString都返回[object Object]
"Boolean Number String Function Array Date RegExp Object Error".split(" ").map(function(item, index) {
    classTypes["[object " + item + "]"] = item.toLowerCase();
});

// 实现类似typeof返回数据类型，支持判断Boolean Number String Function Array Date RegExp Object Error等类型
function type(obj) {
    // null和undefined在这儿区分
    if(obj == null) {
        return obj + "";
    }
    return typeof obj === "object" || typeof obj === "function" ? classTypes[Object.prototype.toString.call(obj)] || "object" : typeof obj;
}
function isWindow(obj) {
    if(obj != null && obj.window === obj) {
        return true;
    }
    return false;
}
function isArrayLike(obj) {
    // obj必须有length属性，把obj.length赋值给length
    var length = !!obj && "length" in obj && obj.length;
    // type和isWindow前面代码都有定义
    var typeRes = type(obj); 
    if(typeRes === "function" || isWindow(typeRes)) {
        return false;
    }
    // 三个条件满足其一即返回true, 1. 类型是array  2. length等于0(处理arguments类数组) 3. length是数字且大于0 且最后一个属性值是存在的
    return typeRes === "array" || length === 0 || typeof length === "number" && length > 0 && (length -1) in obj;
}
function fn() {
    console.log(arguments);
    console.log(isArrayLike(arguments));
}
fn();

# 上述判断第三点，为什么数组的最后一个元素一定要存在？

先看下面代码示例：

function isArrayLike(obj) {
    // obj 必须有 length属性
    var length = !!obj && "length" in obj && obj.length;
    var typeRes = type(obj);

    // 排除掉函数和 Window 对象
    if (typeRes === "function" || isWindow(obj)) {
        return false;
    }
    if(typeRes === "array") { // 是数组
        return true;
    }
    if(length === 0) { // 无参数的类数组 {length:0}
        return true;
    }
    if(typeof length === "number" && length > 0 && (length - 1) in obj) { // 有参数的类数组
        return true;
    }
}
var arr = {  // 这是一个正确的类数组对象， 第三个条件是为了处理这种类数组对象
    0: 1,
    length: 1
}
console.log(isArrayLike(arr));  // true
