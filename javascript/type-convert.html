// 实现Boolean函数，记住number类型三个值（+0,-0,NaN）
function Boolean(param) {
    var result = [undefined, null, +0, -0, NaN, "", false];
    if(param === undefined || result.includes(param)) {
        return false;
    } else {
        return true;
    }
};

// 实现Number构造函数
function Number(input) {
    // 1. 判断是否原始值
    if(isPrimitiveValue(input)) {
        return ToNumber(input);  // 2. 原始值转数字
    } else { // 非原始值，对象
        // 3. ToPrimiive把对象转换为原始值，转不了就抛异常
        var value = ToPrimitive(input, "number");
        // 4. ToPrimitive转原始值后，再把原始值转数字
        return ToNumber(value);
    }
}
// 把对象转换为原始值
function ToPrimitive(input, PreferredType) {
    if(PreferredType === "number") {
        if(isPrimitiveValue(input)) { // 如果是原始值，直接返回
            return input;
        }
        // 3.1 先调用valueOf方法试着把input转为原始值
        input = input.valueOf();
        if(isPrimitiveValue(input)) {
            return input;
        }
        // 3.2 在调用toString方法试着吧input转换为原始值
        input = input.toString();
        if(isPrimitiveValue(input)) {
            return input;
        }
        // 3.3 转不了，抛出异常
        throw new Error("Uncaught TypeError~")
    }
}

// 原始值转数字
function ToNumber(input) {
    if(arguments.length === 0) { // 不传参
        return +0;
    } else {  // 有参数
        if(input === undefined) {  // Number(undefined)
            return NaN;
        } else if(input === null) {  // Number(null)
            return  +0;
        } else if(typeof input === "boolean") { // Number(true)
            return !!input ? 1 : 0;
        } else if(typeof input === "number") { // Number(1)
            return input;
        } else if(typeof input === "string") { // 最复杂的场景， Number("001 ")
            if(input.charAt(0) === "") {
                input = input.replace(/\s+/, "");
            }
            return stringToNumber(input);  // 字符串转数字
        }
    }
}
// 原值值字符串转数字
// 一个一个字符找，去掉前导空格和0或者16等机制数，遇到字符串直接返回NaN
function stringToNumber(str) {
    var length = str.length;
    var count = 0, 
            res, 
            tipCount = 0,   // 记录点的个数
            isTip,        // 是不是点（.）
            firstChar = str.charAt(0); // 首字母
    do {
        res = str.charCodeAt(count);  // 获取字母的unicode编码
        if(count === length) {
            return +str;
        }
        if(count === 0 && firstChar === "-") { // 首字母是"-"
            count++;
            continue;
        }
        if(str.substr(0, 2) === "0x" || str.substr(0, 2) === "0X") { // 16进制
            count +=2;  
            continue;
        }
        
        isTip = str.charAt(count) === ".";
        if(isTip) {
            tipCount++;
        }

        if((48 <= res && res <=57 || isTip )&& tipCount <=1) { // 遇到数字继续往后找
            count++;
        }  else {  // 只要有一个非数字，直接返回NaN
            return NaN;
        }
    } while (count <= length);
}
// typeof判断是否原始值
function isPrimitiveValue(input) {
   if(input ===undefined || input === null || typeof input === "boolean" || typeof input === "string" || typeof input === "number") {
       return true;
   } else {
       return false;
   }
}

Number构造函数传入非字符串可以参考上图 如果Number构造函数传入字符串，会被试图转换成一个整数或浮点数，而且会忽略所有的前导0，如果有一个字符不是数字，结果都会返回NaN

parseInt和parseFloat会尽可能解析更多的数值

// 实现String构造函数
function String(input) {
    // 1. 判断是否为原始值
    if(isPrimitiveValue(input)) {
        return ToString(input);      // 实际是底层实现，并未暴露在Js中
    } else { // 对象
        // 2. 把对象转换为原始值
        var value = ToPrimitive(input, "string");
        return ToString(input);  // 原始值转字符串
    }
}
// 把对象转换为原始值
function ToPrimitive(input, PreferredType) {
    if(PreferredType === "string" || (PreferredType === undefined && Object.prototype.toString.call(input) === "[object Date]")) { // 字符串或者日期+类型未传
        if(isPrimitiveValue(input)) { // 原始值直接返回
            return input;
        }
        // 先调用toString试着把input转换为原始值
        input = input.toString();
         if(isPrimitiveValue(input)) { // 原始值直接返回
            return input;
        }
        // 再调用valueOf试着把input转换为原始值
        input = input.valueOf();
         if(isPrimitiveValue(input)) { // 原始值直接返回
            return input;
        }
        throw new Error("Uncaught TypeError~");
    }
}
// 原始值转字符串
function ToString(input) {
    if(arguments.length === 0) { // 没有传参
        return "";
    } else { // 有参数
        if(input === undefined) { // String(undefined)
            return "undefined";
        } else if(input === null) { // String(null)
            return "null";
        } else if(typeof input === "boolean") {  // String(true)
            return !!input ? "true" : "false";
        } else if(typeof input === "string") {  // String("abc")
            return input;
        } else if(typeof input === "number") { // String(1)
            return "" + input;
        }
    }
}

   }

# JSON.stringify

JSON.stringify方法可以把一个javascript值转换为json字符串， 实际是调用如下ToString方法

// 原始值转字符串
function ToString(input) {
    if(arguments.length === 0) { // 没有传参
        return "";
    } else { // 有参数
        if(input === undefined) { // String(undefined)
            return "undefined";
        } else if(input === null) { // String(null)
            return "null";
        } else if(typeof input === "boolean") {  // String(true)
            return !!input ? "true" : "false";
        } else if(typeof input === "string") {  // String("abc")
            return input;
        } else if(typeof input === "number") { // String(1)
            return "" + input;
        }
    }
}
  1. 处理基本类型时，与ToString方法基本相同，除了undefined
console.log(JSON.stringify(null));      // null
console.log(JSON.stringify(undefined));      // undefined，不是字符串
console.log(JSON.stringify(true));      // true
console.log(JSON.stringify(12));      // 12
console.log(JSON.stringify("abc"));      // "abc"
  1. 布尔值，数字，字符串的包装对象会在序列化话过程中转换为原始值
console.log(JSON.stringify([Boolean(false), new Number(1), new String("a")]));          // [false, 1, "a"]
  1. undefined, 函数已经symbol值， 在序列化过程中，数组中会被转换为null, 在非数组中会被忽略
console.log(JSON.stringify([undefined, Object, Symbol("abc")]));  // [null, null, null]
console.log(JSON.stringify({x: undefined, y: Object, z: Symbol("abc")})) // {}
  1. JSON.stringify第二个参数replacer， 可以是数组或者函数，用来指定序列化过程中需要处理的属性
console.log(JSON.stringify({name: "alan", age: 20, sex: "man"}, replacer)); // {"age":20}
function replacer(key, value) {
    if(typeof value === "string") {
        return undefined
    } else {
        return value;
    }
}
console.log(JSON.stringify({name: "alan", age: 20, sex: "man"}, ["name", "age"])); // {"name":"alan","age":20}
  1. 如果被序列化的对象拥有了toJSON方法，会覆盖默认序列化的方法
var obj = {
    name: "hello",
    toJSON: function() {
        return "world";
    }
}

console.log(JSON.stringify(obj)); // "world"
