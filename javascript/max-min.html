# 如何求数组的最大值和最小值

# 求数组最大值

用法：返回一组数中的最大值，代码示例如下：

// 数组最大值
console.log(Math.max(5, 3, 2, 8));  // 8
console.log(Math.max(5, 3, 2, 8, NaN));  // NaN
console.log(Math.max(1, undefined));  // NaN , undefined转数值为NaN
console.log(Math.max(1, {}));  // NaN , {}转数值为NaN
console.log(Math.max());  // -Infinity

var max = Math.max();  // -Infinity
var min = Math.min();   // Infinity
console.log(min > max); // true

结论： 1. 有任意参数不能被转换为数值，则结果返回NaN 2.没有参数返回-Infinity(负无穷大) 3. Math.min()大于Math.max()

# 手动实现求数组最大值

# for循环比较

for循环两两比较，把两者大的值保存起来

 //思路1：for循环比较，把两个数中大的值赋值result
function max(arr) {
    var length = arr.length || 0;
    var result = arr[0];
    for (let i = 1; i < length; i++) {
        if(result < arr[i]) {
            result = arr[i];
        }
    }
    return result;
}
var arr = [6, 4, 1, 8, 2, 11, 23];
console.log(max(arr)); // 23

# for + Math.max

和2.1类似，通过Math.max获取较大值

// 思路2：for + Math.max, 循环遍历所有的值，依次比较取得大的数值
function max2(arr) {
    var length = arr.length || 0;
    var result = arr[0];
    for (let i = 1; i < length; i++) {
        result = Math.max(result, arr[i]);
    }
    return result;
}
var arr = [6, 4, 1, 8, 2, 11, 23];
console.log(max2(arr)); // 23

# reduce + Math.max

遍历数组获得一个最终值就可以使用reduce

 // 思路3： reduce, 遍历数组获得一个最终值就可以使用reduce
function max3(arr) {
    return arr.reduce(function(prev, next){
        return Math.max(prev, next);
    })
}
var arr = [6, 4, 1, 8, 2, 11, 23];
console.log(max3(arr)); // 23

# sort排序

sort排序（升序），最后一个值就是最大值。

// 思路4： sort排序, 升序，最后一个值就是最大值
var arr = [6, 4, 1, 8, 2, 11, 23];
var arr1 = arr.concat().sort((a, b)=>a - b);  // a-b 正序
console.log(arr1[arr1.length-1]); // 23

# eval + Math.max

// 思路5：eval + Math.max
function max4(arr) {
    return eval("Math.max(" + arr + ")");  //eval中arr会自动调用toString
}
var arr = [6, 4, 1, 28, 2, 11, 23];
console.log(max4(arr)); // 28

# Math.max + apply

// 思路6： apply, 第二个参数是数组
function max5(arr) {
    return Math.max.apply(null, arr);
}
var arr = [6, 4, 1, 8, 2, 11, 23];
console.log(max5(arr)); // 23

# Math.max + 扩展运算符（...）

// 思路7：Math.max + ...[1,2,3]
function max6(arr) {
    return Math.max(...arr);   // ...arr 数组变字符串
}
var arr = [6, 4, 1, 28, 2, 11, 23];
console.log(max6(arr)); // 28
