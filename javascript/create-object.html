# 创建对象的多种方式以及优缺点

# 工厂模式

// 工厂模式
function createPerson(name) {
    var obj = new Object();
    obj.name = name;
    obj.getName = function() {
        return obj.name;
    }
    return obj;
}
// 只要传人名，就可以创建一个对象拥有name属性和获取name属性的方法
// 缺点： 对象无法识别（无法知道是哪种对象类型），所有实例都指向一个原型（因为没有修改原型）

# 构造函数模式

    // 模式一
  function Person(name) {
        this.name = name;
        this.getName = function() {
            return this.name;
        };
    }
    var person1 = new Person("alan1");

优点：实例对象可以识别为一个特定的类型 缺点： 每次创建实例，每个方法（getName）都要被创建一次

    // 模式二， 在模式一基础上优化，把getName方法提取出来
     function Person(name) {
        this.name = name;
        this.getName = getName;  // 把方法提取出來，每次创建实例方法不会被重复创建（函数创建需要开辟内存空间，耗性能），这种写法只是指向了函数内存地址。
    }
    function getName() {  // 函数放在外部，不会被多次创建
        console.log(this.name);
    }

优点：方法不会被多次创建 缺点：体现不出封装性，还新增了一个全局方法

# 原型模式

    // 原型模式一
    function Person(name) {
        
    }
    Person.prototype.name = "alan"; // 实例可以访问原型上的属性跟方法
    Person.prototype.getName = function() {
        console.log(this.name);
    }
    var person = new Person("alan");

    // 原型模式二, 只要说明重写原型需要constructor重新指向构造函数
   function Person(name) {
            
    }
    Person.prototype = {
        constructor: Person,   //原型被重新复制，constructor需要重新指向到构造函数
        username : "alan",
        getName: function() {
            console.log(this.username);
        }
    }
    var person = new Person();

优点：方法不会被重建， 缺点：1. 所有实例的属性和方法都共享(希望name属性可以私有，所以组合模式会放到构造函数中)，2.不能初始化参数（无法自定义name属性）

结论： 给实例共享的放原型（Person.prototype）上，私有属性放构造函数中，组合模式就是这么玩的。

# 组合模式

    // 构造函数（私有属性和方法）和原型（共享属性和方法）模式组合
    function Person(name) {
        this.name = name;
    }
    Person.prototype = {
        constructor: Person,   //原型被重新复制，constructor需要重新指向到构造函数
        getName: function() {
            console.log(this.username);
        }
    }
    var person = new Person();

优点： 该共享的共享（原型对象上），该私有的私有（构造函数上），使用最广泛的方式 缺点：更好的封装性

    // 更好的封装性
     function Person(name) {
        this.name = name;
        if(typeof this.getName != "function") {
            Person.prototype.getName = function() {
                console.log(this.name);
            }
        }
    }
    var person1 = new Person("alan1");  
    var person2 = new Person("alan2");  
    person1.getName();  // alan1
    person2.getName();  // alan2

function Person(name) {
    this.name = name;
    if(typeof this.getName != "function") {
        Person.prototype ={
            constructor: Person,
            getName:  function() {
                console.log(this.name);
            }
        }
    }
}
var person1 = new Person("alan1");  
var person2 = new Person("alan2");
person1.getName();  // person1.getName is not a function  为什么报错？ 
person2.getName();

# 寄生构造模式

 function Person(name) {
    var obj = new Object();
    obj.name = name;
    obj.getName = function() {
        console.log(this.name);
    }
    return obj;
}
var person = new Person("alan");
console.log(person instanceof Person);  // false
console.log(person instanceof Object);  // true

结论：寄生构造函数比工厂模式在创建对象时，多使用了一个new

# 稳妥构造函数模式

 function Person(name) {
    var obj = new Object();
    obj.getName = function() {
        console.log(this.name);
    }
    return obj;
}

与寄生构造函数的区别：

  1. 新创建的实例调用的方法不引用this
  2. 不适用new操作符调用构造函数
