# 细数继承的多种方式和优缺点

# 原型链继承

function Parent() {
    this.name = ["alan"];
}
Parent.prototype.getName = function() {
    console.log(this.name);
}
function Child() {

}
Child.prototype = new Parent();
var child1 = new Child();
var child2 = new Child();
child1.name.push("alan1");
console.log(child2.getName());   // ["alan", "alan1"]

问题：

  1. 引用类型的属性被所有实例共享 // 实例化时需要在构造函数绑定this
  2. 创建Child实例时，不能像Parent传参 // 在创建Child实例时，需要调用Parent方法传参

# 经典继承

解决原型继承的两个问题

function Parent(name) {
    this.name = ["alan"];
    this.getName = function() {  // 方法都在构造函数中定义，每次创建实例都需要创建一次方法
        console.log(this.name);
    }
}
function Child(name) {
    Parent.call(this, name);  // 把name挂载在了Child实例上
}
Child.prototype = new Parent();
var child1 = new Child();
var child2 = new Child();
child1.name.push("alan1");
console.log(child2.getName());   // ["alan"]

结论：解决了原型链继承的两个问题

  1. 避免了引用类型属性被所有实例共享
  2. 可以在Child中像Parent传参

缺点： 方法都在构造函数中定义（），每次创建实例都会一遍方法

# 组合继承

原型链继承和经典继承双剑合璧

  1. 私有属性和方法定义在构造函数中
  2. 公有属性和方法定义在构造函数原型上
  3. 子类的原型指向父类的实例（所以既能访问父类构造函数中的属性和方法，也能访问父类原型中的属性和方法）
 function Parent(name) {
    this.name = name;
    this.colors = ["red"]
}
Parent.prototype.getName = function() {
    console.log(this.colors)
}
function Child(name) {
    Parent.call(this, name);   // 实例化第一次调用
}
Child.prototype = new Parent();  // 实例化第二次调用
var child1 = new Child("alan1");
child1.colors.push("green");
console.log(child1.colors);
var child3 = new Child("alan3");
console.log(child3.colors);

缺点：组合模式最大的缺点，就是每次实例化Child要调用两次Parent

# 原型式继承

将传入的对象作为构建对象的原型，类似Object.create

 function createObject(o) {
    function F() {};
    F.prototype = o;
    return new F();
}

缺点：包含引用类型的属性值始终都会共享相应的值，这点跟原型链一样，如下代码所示

 function createObject(o) {
    function F() {};
    F.prototype = o;
    return new F();
}
var person = {
    name: "alan",
    colors: ["red", "green"]
}
var p1 = createObject(person);
p1.colors.push("yellow");
var p2 = createObject(person);
console.log(p2.colors); // ["red", "green", "yellow"]

# 寄生式继承

  function createObj(o) {
    var clone = Object.clone(o);  // clone是一个实例
    clone.sayName = function() {
        console.log("hello world")
    }
    return clone;
}

结论：实例上添加一个方法，再把实例返回，所以方法是寄生在实例上的 缺点：每次创建实例都要创建一次方法（与借用构造函数类似）

# 寄生组合式继承

组合继承最大的缺点是每次实例化都要调用两次构造函数(Parent) 参考组合继承的代码 一次是设置子类型的原型时

    Child.prototype = new Parent();

一次是创建子类型实例时

    var child1 = new Child("alan");  // 在构造函数中

组合继承

function Parent(name) {
    this.name = name;
    this.colors = ["red"]
}
Parent.prototype.getName = function() {
    console.log(this.colors)
}
function Child(name) {
    Parent.call(this, name);   // child1实例上会挂载一个colors属性
}
Child.prototype = new Parent();  // Child.prototype对象上会挂载一个colors属性
var child1 = new Child("alan1");
console.log(child1);
结论：child1对象中，Child.prototype和child1都有一个colors属性 
function Parent(name) {
    this.name = name;
    this.colors = ["red"]
}
Parent.prototype.getName = function() {
    console.log(this.colors);
}
function Child(name, age) {
    Parent.call(this, name);  
    this.age = age;
}
function F() {}
F.prototype = Parent.prototype; 
Child.prototype = new F();  // // 没有new Parent（没有执行Parent.call(this, ...)）了, Parent实例上不会挂载colors
var child1 = new Child("alan1");
console.log(child1)
