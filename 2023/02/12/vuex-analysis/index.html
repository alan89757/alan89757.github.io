# 介绍 Vuex

Vuex 是一个专为 Vue.js 应用程序开发的状态管理模式，在根组件注入 Store 实例，共享组件状态。

# 安装

// CDN
<script src="vue.js"></script>
<script src="vuex.js"></script>

// NPM
npm install vuex --save

# 简单案例

// 实例化Store
var store = new Vuex.Store({
  state: {
    count: 0,
  },
  mutations: {
    increment(state) {
      state.count++;
    },
  },
});
// store实例全局注册
var vm = new Vue({
  el: "#app",
  store, // store实例注入
  computed: {
    count() {
      return this.$store.state.count;
    },
  },
  methods: {
    add() {
      this.$store.commit("increment");
    },
  },
  template: `<div>
    <p>{{count}}</p>  
    <button @click='add'>add</button>
  </div>`,
});

# Vuex.Store 可选项

四个可选项属性： state, getters, actions, mutations

# state

// $store.state获取数据方式：
var vm = new Vue({
  el: "#app",
  computed: {
    // 获取state数据，$store.state
    // 方式一
    count() {
      return this.$store.state.count;
    },
    // 方式二
    ...mapState([
      "count", // this.$store.state.count的语法糖
    ]),
    // 方式三
    ...mapState({
      count: (state) => state.count,
      // countAlias: "count", // 等同于state => state.count
    }),
  },
});

# getters

// 通过属性
// $store.getters获取数据方式：
var vm = new Vue({

    count() {
      return this.$store.getters.count;

// 通过方法
var store = new Store({
  state: {
    todos: [
      { id: 1, text: "...", done: true },
      { id: 2, text: "...", done: false },
    ],
  },
  getters: {
    // getters方法
    getTodoById: (state) => (id) => {
      return state.todos.find((todo) => todo.id === id);
    },
  },
});
// 使用方式
store.getters.getTodoById(2);

# actions

var store = new Store({
  actions: {
    actionA({ commit }) {
      return new Promise((resolve, reject) => {
        setTimeout(() => {
          commit("someMutation");
          resolve();
        }, 1000);
      });
    },
  },
});
// 使用
store.dispatch("actionA").then(() => {
  // ...
});

var store = new Vue({
  el: "#app",
  store, // store注入Vue可选项，挂载到this.$store
  computed: {
    count() {
      return this.$store.state.count;
    },
  },
  methods: {
    // action 方式一
    addCount() {
      this.$store.dispatch("addCount"); // 分发action
    },
    // action 方式二
    ...mapActions([
      "addCount", // 将 `this.addCount()` 映射为 `this.$store.dispatch('addCount')`
    ]),
    // action 方式三
    ...mapActions({
      add: "addCount", // 将 `this.add()` 映射为 `this.$store.dispatch('addCount')`
    }),
    // 异步处理
    addCountAsync(context) {
      setTimeout(() => {
        // 处理异步请求
        context.commit("increment"); // 异步分发mutations
      }, 2000);
    },
  },
  template: `<div>
    <h1>hello</h1>
    <p>count: {{count}}</p>
    <button @click='add'>addCount</button>
  </div>`,
});

# modules

const moduleA = {
  namespaced: true,
  state: {
    countA: 10,
  },
  getters: {
    getCountA() {
      return this.$store.state.countA;
    },
  },
  actions: {
    increment(context) {
      context.commit("increment");
    },
  },
  mutations: {
    increment(state) {
      state.countA = "a-" + state.countA;
    },
  },
};
const moduleB = {
  namespaced: true,
  state: {
    countB: 11,
  },
  getters: {
    getCountB() {
      return this.$store.state.countB;
    },
  },
  actions: {
    increment({ dispatch, commit, getters, rootGetters }) {
      commit("increment");
      dispatch("someOtherAction", null, { root: true }); // 分发全局action
    },
  },
  mutations: {
    increment(state) {
      state.countB = "b-" + state.countB;
    },
  },
};
var store = new Vuex.Store({
  modules: {
    a: moduleA,
    b: moduleB,
  },
});

var vm = new Vue({
  el: "#app",
  store,
  computed: {
    countA() {
      return this.$store.state.a.countA;
    },
    countB() {
      return this.$store.state.b.countB;
    },
  },
  methods: {
    addA() {
      this.$store.dispatch("a/increment");
    },
    addB() {
      this.$store.dispatch("b/increment");
    },
  },
  template: `<div>
    <h1>hello</h1>
    <p>countA: {{countA}}</p>
    <p>countB: {{countB}}</p>
    <button @click='addA'>addA</button>
    <button @click='addB'>addB</button>
  </div>`,
});

  computed: {
    ...mapState("some/nested/module", {
      a: (state) => state.a,
      b: (state) => state.b,
    }),
  },
  methods: {
    // 第一个参数为上下文
    ...mapActions("some/nested/module", [
      "foo", // -> this.foo()
      "bar", // -> this.bar()
    ]),
  },
});

// 方式二
import { createNamespacedHelpers } from "vuex";
// 创建基于「some/nested/module」命名空间辅助函数
const { mapState, mapActions } = createNamespacedHelpers("some/nested/module");

export default {
  computed: {
    // 在 `some/nested/module` 中查找
    ...mapState({
      a: (state) => state.a,
      b: (state) => state.b,
    }),
  },
  methods: {
    // 在 `some/nested/module` 中查找
    ...mapActions(["foo", "bar"]),
  },
};

# 实现一个简单 Vuex

// index.html
const { Store } = Vuex;
var store = new Store({
  state: {
    count: 0,
    todos: [
      { id: 1, name: "todo1", done: true },
      { id: 2, name: "todo2", done: false },
    ],
  },
  getters: {
    filterTodos() {
      return this.$store.state.todos.filter((item) => item.done);
    },
  },
  actions: {
    increment(store) {
      store.commit("increment");
    },
  },
  mutations: {
    increment(state) {
      state.count++;
    },
  },
});
var vm = new Vue({
  el: "#app",
  store,
  computed: {
    count() {
      return store.state.count;
    },
    filterTodos() {
      return store.getters.filterTodos;
    },
  },
  methods: {
    add() {
      // store.commit('increment');
      store.dispatch("increment");
    },
  },
  template: `<div>
        <p>{{count}}</p>  
        <p v-for="item in filterTodos" :key="item.id">{{item.name}}</p>  
        <button @click='add'>add</button>
      </div>`,
});

// vuex.js
(function (global, factory) {
  global.Vuex = factory();
})(window, function () {
  let Vue;
  // 实现Store类
  class Store {
    constructor(options) {
      if (!Vue && typeof window !== "undefined" && window.Vue) {
        install(window.Vue);
      }
      // 保存用户配置
      this._mutations = options.mutations || {};
      this._actions = options.actions || {};
      this._wapperGetter = options.getters || {};

      // 定义计算属性
      const computed = {};
      this.getters = {};
      // 绑定this
      const store = this;

      Object.keys(this._wapperGetter).forEach((key) => {
        const fn = this._wapperGetter[key];

        // 同名方法执行函数并传入state
        computed[key] = function () {
          return fn.call(store._vm, store.state);
        };

        // 给getters判断只读的属性
        console.log("store.getters--", store.getters);
        Object.defineProperty(store.getters, key, {
          get: () => {
            return store._vm[key];
          },
        });
      });

      // 实现响应式
      this._vm = new Vue({
        data: {
          $$state: options.state,
        },
        computed,
      });

      this.commit = this.commit.bind(this);
      this.dispatch = this.dispatch.bind(this);
    }

    get state() {
      return this._vm._data.$$state;
    }

    set state(v) {
      console.error("please use replaceState to reset state");
    }

    commit(type, payload) {
      const entry = this._mutations[type];

      if (entry) entry(this.state, payload);
    }

    dispatch(type, payload) {
      const entry = this._actions[type];

      if (entry) entry(this, payload);
    }
  }
  // 实现插件
  function install(_Vue) {
    Vue = _Vue;
    Vue.mixin({
      beforeCreate() {
        if (this.$options.store) {
          Vue.prototype.$store = this.$options.store;
        }
      },
    });
  }
  var index_cj = {
    Store,
    install,
  };
  return index_cj;
});
