本文主要介绍vue-router的使用场景和vue-router的原理分析

# 1. 搭建基础路由

// main.js
import Vue from "vue";
import App from "./App.vue";
import router from "./router/router";
new Vue({
  el: "#app",
  router,
  render: (h) => h(App),
});

// router.js
import Vue from "vue";
import VueRouter from "vue-router";
Vue.use(VueRouter);
const routes = [
  {
    path: "/home",
    component: Home,
  }
];
export default new VueRouter({
  mode: "hash",
  routes,
});

// webpack.config.js
{
  module: {
    rules: [
      {
        test: /\.vue$/,
        use: 'vue-loader'
      },
    ]
  },
  resolve: {
    alias: {
      vue: 'vue/dist/vue.esm.js'  // 需要请求完整版
    }
  },
}

# 2. 子路由 + 重定向 + 动态参数

const router = new VueRouter({
  mode: 'hash',
  base: __dirname,
  routes: [
    {
      path: '/users',
      component: Users,
      children: [{ path: ':username', name: 'user', component: User }] // 子路由
    },
    {
      path: '/gallery',
      component: Gallery,
      children: [
        {
          path: '',
          name: 'gallery',
          redirect: { name: 'image', params: { imageId: 'image1' }}  // 重定向
        },
        { path: ':imageId', component: Image, name: 'image' }   // 动态参数
      ]
    }
  ]
})

# 3. 权限验证 + 登录 + 登出

// 路由拦截
function requireAuth(to, from, next) {
  if(!auth.loggedIn()) { // 未登陆，跳转到等登陆页
    next({
      path: '/login',
      query: { redirect: to.fullPath }
    })
  } else {
    next(); 
  }
}
const router = new VueRouter({
  mode: 'hash',
  routes: [
    { path: '/about', component: About },
    { path: '/login', component: Login }, // 登陆页
    { path: '/dashboard', component: Dashboard, beforeEnter: requireAuth },  // 路由前权限校验
    { path: '/logout', beforeEnter(to, from, next) {  // 登出，删除登陆状态
      auth.logout();
      next('/')
    }}
  ]
});

// auth.js
export default {
  loggedIn () { // 是否登录
    return !!localStorage.token;
  },
  signIn(token) { // 登录
    localStorage.token = token;
  },
  logout(cb) {
    delete localStorage.token;
    if(cb) cb();
  }
}


<!-- login.vue -->
<template>
  <div>
    <p><input v-model="username"/></p>
    <button @click="signIn">登陆</button>
    <button @click="logout">登出</button>
  </div>
</template>
<script>
import auth from './auth.js'
export default {
  data() {
    return {
      username: ''
    }
  },
  methods: {
    signIn() {
      auth.signIn(this.username);
      this.$router.replace(this.$route.query.redirect || '/')
    },
    logout() {
      next('/logout')
    }
  }
}
</script>

Vue Router的导航方法(push、replace、go)在各类路由模式(history、hash和abstract)下表现一致。
