# 实现防抖和节流

# 使用场景：频繁的事件触发，比如：

  1. window.resize, window.scroll
  2. mousedown, mousemove
  3. keyup, keydown

原理：一定在事件触发n秒后才执行，如果在事件触发的n秒内又触发了这个事件，那就以新事件的时间为准，n秒之后才执行。

<div id="container" style="width: 100%; height: 200px; background-color: aqua;"></div>
<script>
    var count =0;
    function getUserAction(e) {
        // console.log(this);
        console.log(e);
        container.innerHTML = ++count;
    }
    function debounce(fn, wait) {
        var timer,    // 定时器的指针
                result;  // 执行返回值
        return function() {  // 1. 该匿名函数才是事件绑定函数，需要保存this指向上下文和传参
            var context = this;  // 2. 改变用户自定义函数的this指向上下文
            var args = arguments;  // 3. 传参，把事件对象传递过去
            clearTimeout(timer);
            timer = setTimeout(() => {
                result = fn.apply(context, args);
            }, wait);
            return result;
        }
    } 
    // 事件绑定
    var container = document.getElementById("container");
    container.onmousemove = debounce(getUserAction, 3000);                                                                                                                                             
</script>

# 行

需求：我希望事件触发立即执行函数（比如：在输入框粘贴内容） 解决：新增一个immediate参数，true表示事件触发立即执行

function debounce(fn, wait, immediate) {
    var timer,    // 定时器的指针
            result;  // 执行返回值
    return function() {  // 1. 该匿名函数才是事件绑定函数
        var context = this;  // 2. 改变用户自定义函数的this指向
        var args = arguments;  // 3. 传参，把事件对象传递过去
        if(timer) { clearTimeout(timer); } // timer指针存在再删除
        if(immediate) { // true，表示事件触发立即执行
            var callNow = !timer;  // timer存在，说明事件函数已经触发过了
            callNow ? fn.apply(context, args) : ""; // 不存在，立即调用，同时设置在wait时间后清除定时器
            timer = setTimeout(() => {
                timer = null;  // 手动在wait时间后清除定时器
            }, wait);
        } else {
            timer = setTimeout(() => {
                result = fn.apply(context, args);
            }, wait);
        }
        
        return result;
    }
}

# 节流

原理: 如果持续触发某个事件，每隔一段时间内，只执行一次该事件

# 有两种方式实现节流：

# 1. 使用时间戳

 // 节流-时间戳
function throttle(fn, wait) {
    var previous = 0,
        result,
        timer;
    return function() {
        var  context = this,
                args = arguments,
                now = +new Date();
        if(wait < now -previous) { // 事件函数已执行过
            previous = now;
            result = fn.apply(context, args);
        }
        return result;
    }
}

# 2. 使用定时器

// 节流-定时器，定时器不存在就设置定时器，已存在就不设置
function throttle(fn, wait) {
    var result,
            timer;
    return function() {
        var  context = this,
                args = arguments;
        if(!timer) {  // timer空指针就设置定时器
            timer = setTimeout(() => {
                timer = null;  // 执行任务立即清除定时器
                result = fn.apply(context, args);
            }, wait);
        }
        return result;
    }
}

比较以上时间戳和定时实现节流：

  1. 时间戳实现节流能立即执行（因为前一次执行时间为0），定时器实现的节流会在n秒后第一次执行（因为事件触发时才设置定时器）
  2. 时间戳实现节流在停止触发后没有办法执行事件函数（没有定时器任务），定时器实现的节流会在停止触发后再执行一次事件（因为事件触发设置了定时器任务）

# 综合时间戳和定时器实现节流优化

需求：想要事件触发立即执行（时间戳），停止触发后还能执行一次（设置定时器任务）

// 节流 时间戳+定时器，定时器不存在就设置定时器，已存在就不设置
function throttle(fn, wait) {
    var result,
            previous = 0,
            timer;
    var throttle =  function() {
        var  context = this,
                args = arguments;
                now = +new Date();
        if(wait < now - previous) {
            if(timer) {  // 执行事件函数了清除下定时器，此次事件触发就能设置定时器任务了
                clearTimeout(timer); 
                timer = null;
            }
            previous = now;
            result = fn.apply(context, args);
        } else if(!timer){ // timer空指针就设置定时器任务
            timer = setTimeout(() => {
                timer = null;  // 执行任务立即清除定时器
                result = fn.apply(context, args);
            }, wait);
        }
        return result;
    }
    return throttle;
}

# 如何确定应该使用防抖还是节流

原理对比： 防抖：在事件触发n秒后才执行，如果在事件触发的n秒内又触发了这个事件，以新事件的时间为准，n秒后再执行 节流：如果持续触发某个事件，每隔一段时间内，只执行一个该事件

结论：防抖以触发事件为基准点，n秒之后执行。节流是持续触发事件以时间间隔为频率，超过时间间隔就执行（类似setInterval）
